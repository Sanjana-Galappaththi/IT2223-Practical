%Example03:
%How can get a multiple outputs by using one function

function [add,sub,mul,div] = operations(a,b)
add=a+b;
sub=a-b;
mul=a*b;
div=a/b;
end

%output: 
 [add,sub,mul,div] = operations(3,2)

     5

     1

    1.5000

%Example 07:
%How to get factorial 

function fact =factorial(n)
fact=1;
for i=1:n
fact = fact *i;
end

%output: 
%inside the Command window you can type =>factorial(5)
ans = 120

%Exampple 08:
%How to get the input for factorial

function fact =factorial()
fact=1;
choice=input('Enter a number');
for i=1:choice
fact = fact *i;
end
end


%Example 09:

function result = myFactorial(n)
    if n == 0  || n == 1
        result = 1;  % Factorial of 0 is 1
    else
        result=n*myFactorial(n-1);
    end
end
